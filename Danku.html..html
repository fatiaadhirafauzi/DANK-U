// Welcome Screen Logic  
function openPortal() {  
document.getElementById('welcome-screen').style.transform = 'translateY(-100%)';  
}
// Tanggal Real-time  
const options = { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' };  
document.getElementById('current-date').innerText = new  
Date().toLocaleDateString('id-ID', options);  
// Afirmasi Acak  
const quotes = [  
'"Belajar tidak pernah melelahkan pikiran."',  
'"Literasi adalah jembatan dari kegelapan menuju cahaya."',  
'"Kamu cukup, kamu berharga, dan kamu mampu."',  
'"Kesalahan adalah bukti bahwa kamu sedang mencoba."',  
'"Setiap kata yang kamu baca adalah investasi untuk dirimu."'  
];  
setInterval(() => {  
const randomQuote = quotes[Math.floor(Math.random() * quotes.length)];  
document.getElementById('quotes').innerText = randomQuote;  
}, 5000);  
// Fitur Curhat  
function kirimCurhat() {  
const input = document.getElementById('chat-input');  
const area = document.getElementById('chat-area');  
if(input.value !== "") {  
const userMsg = document.createElement('div');  
userMsg.innerHTML = `<p style="background: white; padding: 8px; border-radius:  
10px; margin-bottom: 5px; font-size: 0.9rem;">  
area.appendChild(userMsg);  
${input.value}</p>`;  
// Respons Otomatis Beri Semangat  
setTimeout(() => {  
const botMsg = document.createElement('div');  
botMsg.innerHTML = `<p style="color: var(--korean-pink); font-size: 0.8rem;  
margin-bottom: 10px; padding-left: 10px;"><b>DANK U:</b> Terima kasih sudah bercerita.  
Kamu luar biasa!</p>`;  
area.appendChild(botMsg);  
area.scrollTop = area.scrollHeight;  
}, 1000);  
input.value = "";  
}
}
